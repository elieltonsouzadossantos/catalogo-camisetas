  function aplicarFiltros() {
    let lista = produtos.slice();
    const busca = document.getElementById('input-busca').value.toLowerCase();
    const cat   = document.getElementById('filtro-categoria').value;
    const preco = document.getElementById('filtro-preco').value;
    const ordem = document.getElementById('filtro-ordem').value;
    if (busca) lista = lista.filter(function(p){ return p.nome.toLowerCase().includes(busca); });
    if (cat)   lista = lista.filter(function(p){ return p.cat === cat; });
    if (preco) {
      var parts = preco.split('-');
      var min = Number(parts[0]), max = Number(parts[1]);
      lista = lista.filter(function(p){ return p.preco >= min && p.preco <= max; });
    }
    if (ordem === 'menor') lista.sort(function(a,b){ return a.preco - b.preco; });
    if (ordem === 'maior') lista.sort(function(a,b){ return b.preco - a.preco; });
    if (ordem === 'nome')  lista.sort(function(a,b){ return a.nome.localeCompare(b.nome); });
    renderProdutos(lista);
  }
